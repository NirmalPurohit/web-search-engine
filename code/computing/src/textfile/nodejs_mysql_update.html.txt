"yourusername", password: "yourpassword", database: "mydb" }); con.connect(function(err) { if (err) throw err; var sql = "UPDATE customers SET address = 'Canyon 123' WHERE address = 'Valley 345'"; con.query(sql, function (err, result) { if (err) throw err; console.log(result.affectedRows + " record(s) updated"); }); }); Run example » Notice the WHERE clause in the UPDATE syntax: The WHERE clause specifies which record or records that should be updated. If you omit the WHERE clause, all records will be updated! Save the code above in a file called "demo_db_update.js" and run the file: Run "demo_db_update.js" C:\Users\Your Name>node demo_db_update.js Which will give you this result: 1 record(s) updated The Result Object When executing a query, a result object is returned.
